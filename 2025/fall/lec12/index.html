#!/bin/bash
# Ubuntu package installation script

# Function to check if package is installed
is_installed() {
    dpkg -l | grep -q "^ii.*$1 "
}

# Function to install package if not present
install_if_missing() {
    if ! is_installed "$1"; then
        echo "Installing $1..."
        sudo apt update
        sudo apt install -y "$1"
    else
        echo "$1 is already installed"
    fi
}

# Install essential packages
packages=("curl" "wget" "git" "vim" "htop")
for package in "${packages[@]}"; do
    install_if_missing "$package"
done

# Clean up
sudo apt autoremove -y
sudo apt autoclean
#!/bin/bash
# Ubuntu system update script

echo "=== Ubuntu System Update Script ==="
echo "Started at: $(date)"

# Update package lists
echo "Updating package lists..."
sudo apt update

# Show available upgrades
echo "Available upgrades:"
apt list --upgradable

# Ask for confirmation
read -p "Do you want to upgrade? (y/N): " confirm
if [[ $confirm =~ ^[Yy]$ ]]; then
    # Upgrade packages
    sudo apt upgrade -y
    
    # Remove unnecessary packages
    sudo apt autoremove -y
    
    # Clean package cache
    sudo apt autoclean
    
    echo "System update completed successfully!"
else
    echo "Update cancelled by user."
fi

# Check if reboot is required
if [ -f /var/run/reboot-required ]; then
    echo "*** REBOOT REQUIRED ***"
    cat /var/run/reboot-required.pkgs
fi

#!/bin/bash
# Script with desktop notifications

# Check if notify-send is available
if command -v notify-send >/dev/null 2>&1; then
    # Send desktop notification
    notify-send "Script Started" "Ubuntu automation script is running"
fi

# Function for notifications
notify_user() {
    local title="$1"
    local message="$2"
    local urgency="${3:-normal}"  # low, normal, critical
    
    if command -v notify-send >/dev/null 2>&1; then
        notify-send -u "$urgency" "$title" "$message"
    fi
    
    # Also log to terminal
    echo "[$title] $message"
}

# Example usage
notify_user "Backup Started" "Backing up your documents..." "normal"

# Simulate some work
sleep 3

notify_user "Backup Complete" "All files backed up successfully!" "normal"
#!/bin/bash
# Ubuntu system monitoring script

# Configuration
LOGFILE="/var/log/system_monitor.log"
THRESHOLD_CPU=80
THRESHOLD_MEM=90
THRESHOLD_DISK=90

# Function to log with timestamp
log_message() {
    echo "[$(date '+%Y-%m-%d %H:%M:%S')] $1" | sudo tee -a "$LOGFILE"
}

# Check CPU usage
check_cpu() {
    cpu_usage=$(top -bn1 | grep "Cpu(s)" | awk '{print $2}' | cut -d'%' -f1)
    cpu_usage=${cpu_usage%.*}  # Remove decimal part
    
    if [ "$cpu_usage" -gt "$THRESHOLD_CPU" ]; then
        log_message "HIGH CPU: ${cpu_usage}%"
        notify_user "High CPU Usage" "CPU usage is ${cpu_usage}%" "critical"
    fi
}

# Check memory usage
check_memory() {
    mem_usage=$(free | grep Mem | awk '{printf("%.0f", $3/$2 * 100.0)}')
    
    if [ "$mem_usage" -gt "$THRESHOLD_MEM" ]; then
        log_message "HIGH MEMORY: ${mem_usage}%"
        notify_user "High Memory Usage" "Memory usage is ${mem_usage}%" "critical"
    fi
}

# Check disk usage
check_disk() {
    disk_usage=$(df / | tail -1 | awk '{print $5}' | cut -d'%' -f1)
    
    if [ "$disk_usage" -gt "$THRESHOLD_DISK" ]; then
        log_message "HIGH DISK: ${disk_usage}%"
        notify_user "High Disk Usage" "Disk usage is ${disk_usage}%" "critical"
    fi
}

# Main monitoring loop
log_message "System monitoring started"
check_cpu
check_memory
check_disk
log_message "System monitoring completed"
